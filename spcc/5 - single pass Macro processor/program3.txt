MACRO
INCR
MOVER AREG ONE
MOVER BREG TWO
ADD AREG BREG
MEND
MACRO
DNL
MOVER CREG ONE
MOVER DREG TWO
DIV CREG DREG
MEND
MACRO
INCR1
MOVER AREG ONE
MOVER BREG TWO
MULT AREG BREG
MEND
START 200
MOVER AREG FIRST
ADD AREG SECOND
MOVEM AREG RESULT
INCR
PRINT RESULT
RESULT DS 1
FIRST DC 5
INCR1
SECOND DC 7
INCR
DNL
END
